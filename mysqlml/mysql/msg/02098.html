# 賢くないのは許してください。超初心者ですから。^^;

#!/usr/bin/perl

use PNGgraph::lines;

chop($line=<>);
@hanrei = split(//, $line);

while (<>) {
    chop($line=$_);
    @letter = split(//, $line);
    ( $a, $b, $c, $d, $e ) = @letter;
    @d1 = ( @d1, $a );
    @d2 = ( @d2, $b );
    @d3 = ( @d3, $c );
    @d4 = ( @d4, $d );
    @d5 = ( @d5, $e );
}
    @data = ( [@d1], [@d2], [@d3], [@d4], [@d5] );
    $my_graph = new PNGgraph::lines( );
    $my_graph->set(
        bgclr           => white,
        transparent     => 0,
        line_types      => [ 1, 1, 1, 1, 1 ],
        dclrs           => [ black, red, blue, green, aqua ],
        x_label           => 'x-line',
        y_label           => 'y-line',
        title             => 'GRAPH Test(lines)',
        y_min_value       => 0,
        y_max_value       => 100,
        y_tick_number     => 5,
        y_label_skip      => 1
    );

    $my_graph->set_legend(@hanrei);
    $my_graph->plot_to_png ( "line.png", \@data );
