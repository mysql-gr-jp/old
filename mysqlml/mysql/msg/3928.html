  unless( $e{'Body'} =~ /Pass: 12345678\n/){
     return 'from owner only';

     //パスワードをボディから取る処理を入れる
     $e{'Body'} =~s/Pass: 12345678\n//;
  }
#;

＃   unless( $e{'Body'} =~ /^Pass: 12345678\n/){
