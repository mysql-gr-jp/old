mysql> select @i:=2,@x:=@i,@z:=@i;

mysql> select a,@x:=@x+1,if(a<>@x and @z=@i,@z:=@x,null) from table2
where a>@i
 order by a;

| a    | @x:=@x+1 | if(a<>@x and @z=@i,@z:=@x,null) |

mysql> select if(@z<>@i,@z,if(@x=@i,null,@x+1));

| if(@z<>@i,@z,if(@x=@i,null,@x+1)) |

| 6                                 |
