`id` INT( 11 ) NOT NULL AUTO_INCREMENT ,
 `name` VARCHAR( 100 ) DEFAULT NULL ,
PRIMARY KEY (  `id` )
) ENGINE = MYISAM DEFAULT CHARSET = latin1;

INSERT INTO  `test`.`x` 
SELECT * 
FROM  `test`.`a`

・id int autoincrement, not null, PK
・name varchar(100)
の a というテーブルを x というテーブルにコピーした時です。
