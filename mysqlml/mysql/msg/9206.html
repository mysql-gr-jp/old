makeの手順

onigurumaはREADMEに従ってnmakeしておく。

cl -MD -Id:\mysql\include -Id:\arc\oniguruma -c -Tcmregexp.c

type mregexp.def
EXPORTS
mregexp_init
mregexp_deinit
mregexp
mregexp_version

cl -nologo -LD mregexp.obj d:\arc\oniguruma\onig.lib \
 -Femregexp.so -link -incremental:no -opt:ref -def:mregexp.def
