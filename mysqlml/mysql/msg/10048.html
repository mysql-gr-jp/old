mysql> select * from Heikin;

| hiduke     | hyouten |
+------------+---------+
| 2004-08-20 |       4 |
| 2004-08-20 |       5 |

2 rows in set (0.00 sec)

mysql> set @aa='20040820';
Query OK, 0 rows affected (0.02 sec)

mysql> select sum(hyouten)/count(hiduke) from Heikin where hiduke<=@aa;

| sum(hyouten)/count(hiduke) |

|                       4.50 |
+----------------------------+
