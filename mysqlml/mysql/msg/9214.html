mysql>drop table test;

mysql>insert into test (name) values ('あ');

mysql>select * from test;

| name |
+------+
| あ  |
+------+
1 row in set (0.00 sec)

mysql> select length(name) from test;

| length(name) |
+--------------+
|            3 |
+--------------+
1 row in set (0.00 sec)
