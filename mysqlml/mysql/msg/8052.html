-----
$server = "server" # サーバー
$user   = "user"   # ユーザー
$pass   = "pass"   # パスワード
$db     = "db"     # ＤＢ名

$table  = "table"  # 変更したいテーブル名
$col    = "col"    # 変更したいカラム名

require "mysql"

mydb = Mysql.new($server,$user,$pass,$db)

mydb.query("SELECT #{$col} FROM #{$table} ;").each_hash do |myRw|
  if myRw[$col] then
    chs = myRw[$col].sub(/^(.+)@jp-([a-z]).ne.jp/) {
      "#{$1}@#{$2}.vodafone.ne.jp" }
    mydb.query("UPDATE #{$table} SET #{$col} = '#{chs}' WHERE #{$col} = '#{myRw[$col]}' ;")
  end
end
-----
