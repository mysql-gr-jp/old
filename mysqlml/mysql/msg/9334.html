■インストール方法
# tar zxvf mysql-4.1.1-alpha.tar.gz
# cd mysql-4.1.1-alpha
# groupadd mysql
# useradd -g mysql mysql
# passwd mysql
# ./configure --prefix=/usr/local/mysql
  --with-charset=ujis
  --with-extra-charsets=all
# make
# su
# make install
