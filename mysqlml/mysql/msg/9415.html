----------------------------------------------------------------------
	mysql> select * from t where id < 4 order by id desc limit 1;
	+------+------+
	| id   | data |
	+------+------+
	|    3 | bbbb |
	+------+------+

	mysql> select * from t where id > 4 order by id limit 1;
または	mysql> select * from t where id > 4 limit 1;

	+------+------+
	| id   | data |
	+------+------+
	|    9 | dddd |
	+------+------+
