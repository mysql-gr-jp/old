#./configure --prefix=/usr/local/mysql --with-charset=ujis
--with-extra-charsets=all
#make && make install
cp /usr/local/src/mysql/support-files/my-mesium.cnf /etc/my.cnf
#/usr/local/mysql/bin/mysql_install_db

>Fill help tables

>PLEASE REMEMBER TO SET A PASSWORD FOR THE MySQL root USER !

>/usr/local/mysql/bin/mysqladmin -u root -h localhost.localdomain password
'new-password'
>See the manual for more instructions.

>NOTE:  If you are upgrading from a MySQL <= 3.22.10 you should run
>the /usr/local/mysql/bin/mysql_fix_privilege_tables. Otherwise you will not
be
>able to use the new GRANT command!

>You can start the MySQL daemon with:
>cd /usr/local/mysql ; /usr/local/mysql/bin/mysqld_safe &

>You can test the MySQL daemon with the benchmarks in the 'sql-bench'

>cd sql-bench ; perl run-all-tests
