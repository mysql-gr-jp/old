以下、ビット演算｜ＢＯＯＬ演算の例（両方とも同条件）

/* ビット演算 */
CREATE TABLE tbl_bit ( col integer );

SELECT *
  FROM tbl_bit
  WHERE col & 0x11 = 0x11;

/* BOOL演算 */
CREATE TABLE tbl_bool ( col bool[32] );

SELECT *
  FROM tbl_bool
  WHERE col[7] = TRUE AND col[31] = TRUE;
