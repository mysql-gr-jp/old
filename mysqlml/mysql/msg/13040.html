set @r1 = 0;
select scode, namae, max(if(kcode = 4, ktime, null)) as 4P,
max(if(kcode = 9, ktime, null)) as 9P,
max(if(kcode = 13, ktime, null)) as 13P,
@r1:=(@r1 + 1) as rank
