一文の SQL でなくてもよければ、こんな方法でどうでしょうか。

create temporary table tmp (nyuukoSuu int, shukkoSuu int, shouhinmei text);
insert into tmp select kazu, 0, shouhinmei from tableNyuuko;
insert into tmp select 0, kazu, shouhinmei from tableShukko;
select sum(nyuukoSuu), sum(shukkoSuu), shouhinmei from tmp group by shouhinmei;
drop table tmp;
