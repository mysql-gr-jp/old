CREATE TABLE `table1` (
`f_id` bigint(20) unsigned NOT NULL auto_increment,
`f_date_time` date NOT NULL default '0000-00-00',
`aaa` bigint(20) unsigned NOT NULL default '0',
`bbb` bigint(20) unsigned NOT NULL default '0',
`ccc` bigint(20) unsigned NOT NULL default '0',
`ddd` bigint(20) unsigned NOT NULL default '0',
`eee` bigint(20) unsigned NOT NULL default '0',
`fff` bigint(20) unsigned NOT NULL default '0',
PRIMARY KEY (`f_id`),
KEY `f_id` (`f_id`)
) TYPE=MyISAM; 

CREATE TABLE `table2` (
`f_id` bigint(20) unsigned NOT NULL auto_increment,
`f_name` char(201) NOT NULL default '',
`f_code` char(4) NOT NULL default '',
PRIMARY KEY (`f_id`),
KEY `f_id` (`f_id`)
) TYPE=MyISAM;
