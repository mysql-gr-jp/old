	id INT NOT NULL AUTO_INCREMENT,
	username VARCHAR(20) NOT NULL,
	password VARCHAR(20),
	email VARCHAR(50) NOT NULL,
	PRIMARY KEY (id,username)

	INDEX (email) ,
