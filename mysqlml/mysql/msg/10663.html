くらいまでの値が順に入っているだけのものです。
==============================================================
$random_query="select id from test order by rand() limit 10";
$sth=$master_dbh->prepare($random_query);
printf("First\n");
$sth->execute;
while($r=$sth->fetchrow_arrayref){
  printf("%4d\n",$$r[0]);
}

printf("Second\n");
$sth->execute;
while($r=$sth->fetchrow_arrayref){
  printf("%4d\n",$$r[0]);
