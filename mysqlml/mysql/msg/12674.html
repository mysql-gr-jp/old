  COUNT(*) AS Count
FROM
  (SELECT
     *
   FROM
     Recipe_DateOrder
   WHERE
     MATCH( Title_s,Description_s,References_s,Ingredients_s,Tag_s )
     AGAINST( '+"ミ"' IN BOOLEAN MODE )) AS T
