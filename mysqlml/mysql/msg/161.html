void remove_escape(char *name)
{
  char *to;

  for (to=name ; *name ; name++)
  {
      if (*name == '\\' && name[1]){
         name++;
      }
      if( *name >= 0){
         *to++= *name;
      }else{
         *to++= *name;
         *to++= *name;
      }
  }
  *to=0;
}
