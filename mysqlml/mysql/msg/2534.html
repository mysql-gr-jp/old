#include "config.h"

#include <sys/utsname.h>

#ifdef HAVE_SYSCALL_UNAME
int gethostname(char *name, int len)
{
  int ret;
  struct utsname buf, buf2;

  if ((ret = machdep_sys_chroot(&buf)) < OK)
  {
    SET_ERRNO(-ret);
  }
  else
  {
    uname(&buf2);
    strncpy(name,buf2.sysname, len);
  }
  return(ret);
}
#endif
