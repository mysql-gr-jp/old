MySQLは、MySQLのサイトからＳＲＣ．ＲＰＭを拾ってきて
以下のオプションで作りました。

        ./configure \
            --enable-assembler \
            --with-charset=sjis \
            --with-mysqld-user=%{mysqld_user} \
            --with-unix-socket-path=/var/lib/mysql/mysql.sock \
            --prefix=/ \
            --with-extra-charsets=complex \
            --exec-prefix=/usr \
            --libexecdir=/usr/sbin \
            --sysconfdir=/etc \
            --datadir=/usr/share \
            --localstatedir=/var/lib/mysql \
            --infodir=/usr/info \
            --includedir=/usr/include \
            --mandir=/usr/man \
            --with-comment=\"Official MySQL RPM\";

ＰＨＰは、以下です。
./configure --with-config-file-path=/etc/httpd --enable-debugger
 --enable-force-cgi-redirect --enable-mbstring --enable-mbstr-enc-trans
 --with-mysql=/usr

--with-mysql=　は以下を試してみましたがダメでした。。

--with-mysql=/usr
--with-mysql=/usr/lib/mysql
--with-mysql=/

--with-mysql=/usr/lib/mysql
と
--with-mysql=/
だと

とエラーが出ます。
