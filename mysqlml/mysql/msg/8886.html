こんにちは、班石です
ユーザ変数を使う方法で作ってみました。

昇順
select @i:=2,@x:=@i,@z:=@i;
select a , @x:=@x+1 ,if(a<>@x and @z=@i,@z:=@x,null) from table2 where a
> @i order by a;
select if(@z<>@i,@z,if(@x=@i,null,@x+1));
降順
select @i:=10,@x:=@i,@z:=@i;
select a , @x:=@x-1 ,if(a<>@x and @z=@i,@z:=@x,null) from table2 where a
< @i order by a DESC ;
select if(@z<>@i,@z,if(@x=@i,null,@x-1));
