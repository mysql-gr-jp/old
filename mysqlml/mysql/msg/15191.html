`id` int NOT NULL AUTO_INCREMENT,
`num` int NOT NULL DEFAULT '1',
PRIMARY KEY (`id`),
index (`num`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8;

CREATE TABLE `sales` (
`id` int NOT NULL AUTO_INCREMENT,
`num` int NOT NULL DEFAULT '1',
PRIMARY KEY (`id`),
index (`num`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8;

ようなエラーになります。

ALTER TABLE `sales` ADD FOREIGN KEY ( `num` ) REFERENCES `test`.`goods`
(
`num`
