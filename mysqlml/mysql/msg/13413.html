select 顧客テーブル.顧客ID, 顧客テーブル.顧客名,
sum(if(売り上げ is not null and 購入日 between '2006-10-12' and
'2006-10-18', 1, 0)) as

sum(if(購入日 between '2006-10-12' and '2006-10-18', 売り上げ, 0)) as
