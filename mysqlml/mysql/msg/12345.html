-- MySQL dump 9.11
--
-- Host: localhost    Database: test
-- ------------------------------------------------------
-- Server version	4.0.25-nt

--
-- Table structure for table `table1`
--

CREATE TABLE table1 (
  mod int(1) unsigned NOT NULL default '0',
  group int(1) unsigned NOT NULL default '0'
) TYPE=MyISAM;

--
-- Dumping data for table `table1`
--

いた通り`group`としたところ無事にリストアされました。ただし、modは``でく
くらなくてもエラーにはなりませんでした。
