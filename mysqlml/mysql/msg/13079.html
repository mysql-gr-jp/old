> | loginid      | varchar(30)  | YES  |     | NULL    |                |

> | gyoushu      | text         | YES  |     | NULL    |                |
>

> 定しております。

INSERT INTO saller (
    tourokubi,
    loginid,
    gyoutai,
    gyoushu
) VALUES (
    '{$tourokubi}',
    '{$loginid}',
    '{$gyoutai}',
    '{$gyoushu}'
)
__SQL__;
?>
