$ rpmbuild --rebuild perl-DBD-mysql-4.005-1.rf.src.rpm
# rpm -Uvh perl-DBD-mysql-4.005-1.rf.i386.rpm
Preparing...                ########################################### [100%]
   1:perl-DBD-mysql         ########################################### [100%]
