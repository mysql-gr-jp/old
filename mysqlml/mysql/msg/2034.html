> あ、もうついでに画像参照用のCGIものっけちゃいます(笑)。
> とにかく忙しくてすみません(泣)。
> #!/usr/local/bin/perl
> use DBI;
> # <IMG SRC="このCGIおよびItemCodeパラメタ">のようにHTMLから呼び出す
> # 本ソースはjpeg形式のみ対応
>
> # DB接続
> # host,user,passwdは適宜読み替えてください
> $dbh = DBI->connect("DBI:mysql:host","user","passwd");
> $ItemCode = $ENV{'QUERY_STRING'};
> $s = "SELECT ItemPict FROM Items WHERE ItemCode='$ItemCode'";
> $sth = $dbh->prepare($s) || die "Can't Prepare $s:$dbh->errstr\n";
> $rv = $sth->execute || die "Can't execute\n";
> @row = $sth->fetchrow_array;
> print "Content-Type: image/jpeg\n";
> print "Pragma: no-cache\n\n";
> print $row[0];
> $dbh->disconnect;
>
