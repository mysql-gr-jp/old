簡単に以下ではどうでしょうか？
ワンライナーではありませんが。

drop table if exists fri;

create temporary table fri
  select 1st_name as name1, max(height) as heig from friends
  group by name1;

select id, name1, 2nd_name, heig from fri inner join friends
  on fri.name1=friends.1st_name and fri.heig=friends.height
  order by heig desc;
