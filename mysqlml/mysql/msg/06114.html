	private String Disp(String name,String age,String area,String
comment){

		String Area = area;

	private String ConDB(String area,String age,int viewnum){

and ");

		}
		else if( flagArea == true & flagAge == false){

			queryArea	= "";
			queryAge	= ("where age = \'" + age
+"\'");

			queryArea	= ("where area = \'" + area
