  pw_name varchar(32) NOT NULL default ''

  pw_passwd varchar(255) NOT NULL default '',

  pw_dir varchar(255) default NULL,
  pw_shell varchar(255) default '/bin/nosh',

  quota double NOT NULL default '209715200',
