糸瀬と申します。以下でどうでしょうか。
tableAのPRIMARYをidと仮定して記述しています。

SELECT *
FROM tableA
WHERE id IN (
  SELECT tbl.id
  FROM(
      SELECT id AS id
      FROM tableA
      WHERE columnA LIKE '%||カモ||%'
    union all
      SELECT id AS id
      FROM tableA
      WHERE columnA LIKE '%||ガチョウ||%'
    union all
      SELECT id AS id
      FROM tableA
      WHERE columnA LIKE '%||スズメ||%'
  ) AS tbl
  GROUP BY
    tbl.id
  HAVING
    COUNT(tbl.id) >= 2
)
