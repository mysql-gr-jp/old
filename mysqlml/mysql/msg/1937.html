perl CGIの関数。
sub execsql
{

 my($sql_statement);
 ($sql_statement) = @_;

 my $sth = $dbh->prepare($sql_statement) || &error('prep_sql',$0);
 if($dbh) {
  if(defined($sql_statement)) {
   $sth->execute || &error('sql',$0);
  } else {
   &error('sql',$0);
  }
 } else {
  &error('connect');
 }
 $dbh->commit;
 return $sth;

}
